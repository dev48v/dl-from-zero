    A neuron = weighted sum + a threshold
    

    let w = [Math.random(), Math.random()], b = 0;
const sum = x => w[0]*x[0] + w[1]*x[1] + b;
const predict = x => sum(x) >= 0 ? 1 : -1;   // step activation

    2
    The perceptron learning rule
    
        
    

    function trainStep(lr) {
  for (const { x, y } of data) {
    const pred = predict(x);
    if (pred !== y) {                 // only fix mistakes
      w[0] += lr * y * x[0];
      w[1] += lr * y * x[1];
      b    += lr * y;
    }
  }
}

    Repeat until it stops making mistakes
    

    for (let epoch = 0; epoch < 50; epoch++) trainStep(0.1);
// if the data is linearly separable, this is GUARANTEED to converge.
