🔨 A GAN, in code

    
Define a Generator and a Discriminator → sample noise → G makes fakes → train D on real + fake → train G to fool D → loop. ~7 steps. The structure below is the real PyTorch training loop every GAN uses, from toy 2D data to StyleGAN faces.

    
    
      
  1. 1
    Define the two networks (G and D)
    
        
    import torch, torch.nn as nn

# Generator: random noise vector  ->  a fake sample
G = nn.Sequential(
    nn.Linear(latent_dim, 128), nn.ReLU(),
    nn.Linear(128, data_dim),                 # e.g. data_dim = 2 (x, y)
)
# Discriminator: a sample  ->  P(real)  in [0, 1]
D = nn.Sequential(
    nn.Linear(data_dim, 128), nn.ReLU(),
    nn.Linear(128, 1), nn.Sigmoid(),
)
    
          
    2. 

      
  2. 2
    Optimizers + loss (two players, two updaters)
    
        
    # each network has its OWN optimizer — they are trained separately
opt_D = torch.optim.Adam(D.parameters(), lr=2e-4, betas=(0.5, 0.999))
opt_G = torch.optim.Adam(G.parameters(), lr=2e-4, betas=(0.5, 0.999))

bce = nn.BCELoss()          # binary cross-entropy: "is this real?"
real_label, fake_label = 1.0, 0.0
    
          
    3. 

      
  3. 3
    Sample noise → Generator makes fakes
    
        
    # draw a batch of random latent vectors z ~ N(0, 1)
z      = torch.randn(batch, latent_dim)
fakes  = G(z)                       # G turns noise into fake samples
reals  = sample_real_data(batch)    # a real batch from the dataset
    
          
    4. 

      
  4. 4
    Train D — reward catching real + fake
    
        
    # D should output ~1 for reals and ~0 for fakes
opt_D.zero_grad()
loss_real = bce(D(reals),          torch.ones (batch, 1))   # call reals "real"
loss_fake = bce(D(fakes.detach()), torch.zeros(batch, 1))   # call fakes "fake"
loss_D = loss_real + loss_fake
loss_D.backward()                 # .detach() => G is NOT updated here
opt_D.step()
    
          
    5. 

      
  5. 5
    Train G — fool D into saying "real"
    
        
    # freeze nothing; just push G so D thinks the fakes are real (label = 1)
opt_G.zero_grad()
fakes  = G(torch.randn(batch, latent_dim))
loss_G = bce(D(fakes), torch.ones(batch, 1))   # G WANTS D to say "real"
loss_G.backward()                              # gradient flows D -> G
opt_G.step()                                   # only G's weights move
    
          
    6. 

      
  6. 6
    Loop — alternate D then G until equilibrium
    
        
    for step in range(num_steps):
    train_discriminator()     # step 4
    train_generator()         # step 5
    if step % 500 == 0:
        # as G wins, D(fakes) drifts toward 0.5  ->  accuracy ~ 50%
        print(step, "D(real)=", D(reals).mean().item(),
                     "D(fake)=", D(fakes).mean().item())
    
          
    7. 

      
  7. 7
    Generate new samples from the trained G
    
        
    # training done — throw D away, keep G as your generator
G.eval()
with torch.no_grad():
    z       = torch.randn(64, latent_dim)   # fresh noise = fresh, varied outputs
    samples = G(z)                          # brand-new data that looks real
# for images, G is a stack of ConvTranspose2d layers (a DCGAN)
