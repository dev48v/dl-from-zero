    A kernel is a small grid of weights
    const edge = [[-1,-1,-1],
              [-1, 8,-1],
              [-1,-1,-1]];   // sums to 0 → flat areas vanish, edges pop

  2. 2
    Slide it over every pixel
    for (let y = 1; y < h-1; y++)
  for (let x = 1; x < w-1; x++) {
    let sum = 0;
    for (let ky = -1; ky <= 1; ky++)
      for (let kx = -1; kx <= 1; kx++)
        sum += img[y+ky][x+kx] * kernel[ky+1][kx+1];
    out[y][x] = sum;   // one output number per position
  }

  3. 3
    Weight sharing = the big idea
    // the SAME 9 weights are reused at every position.
// a dense layer would need (w·h)×(w·h) weights; a 3×3 kernel needs 9.
// → far fewer parameters, and it detects a feature ANYWHERE in the image.
