🔨 Build a forward pass from scratch

    
No libraries. Just arrays and arithmetic.

    
    
      
  1. 
        
    
          
    1
    
          
    Dot product — the heart of a neuron
    
        
    
        
    Each neuron multiplies its weights by the inputs and sums. That's all w·x is.
    
        
    // weights w = [0.8, -0.5], inputs x = [1.0, 2.0]
const dot = (w, x) => w.reduce((s, wi, i) => s + wi * x[i], 0);
// dot([0.8, -0.5], [1.0, 2.0]) → 0.8*1 + -0.5*2 = -0.2
    
      
    2. 
      
  2. 
        
    
          
    2
    
          
    One neuron = dot + bias + activation
    
        
    
        
    Add a bias b (shifts the result), then pass z through an activation function f.
    
        
    const relu    = z => Math.max(0, z);
const sigmoid = z => 1 / (1 + Math.exp(-z));
const tanh_fn = z => Math.tanh(z);

function neuron(w, b, x, act) {
  const z = dot(w, x) + b;   // z = Wx + b (pre-activation)
  return act(z);              // a = f(z)  (post-activation)
}
    
      
    3. 
      
  3. 
        
    
          
    3
    
          
    A layer = several neurons in parallel
    
        
    
        
    Each neuron in a layer reads the same input vector x. The layer returns one activation per neuron.
    
        
    // W = [[w for neuron0], [w for neuron1], ...]
// b = [b0, b1, ...]
function layerForward(W, b, x, act) {
  return W.map((w, i) => neuron(w, b[i], x, act));
  // returns array of activations — the next layer's input
}
    
      
    4. 
      
  4. 
        
    
          
    4
    
          
    Stack layers — the full forward pass
    
        
    
        
    Chain layerForward calls. The output of one layer is the input of the next.
    
        
    // Network: 2 inputs → 3 hidden (relu) → 1 output (sigmoid)
const W1 = [[0.8,-0.5],[0.1,0.9],[0.3,-0.2]];  // shape 3×2
const b1 = [0.1, -0.1, 0.05];
const W2 = [[0.6, -0.4, 0.7]];                  // shape 1×3
const b2 = [0.0];

function forward(x) {
  const h = layerForward(W1, b1, x, relu);       // hidden: [3]
  const y = layerForward(W2, b2, h, sigmoid);    // output: [1]
  return y[0];
}

forward([1.0, -0.5]);  // → e.g. 0.723
    
      
    5. 
      
  5. 
        
    
          
    5
    
          
    Generalise — any depth, any width
    
        
    
        
    Store layers as an array of {W, b, act} objects and loop.
    
        
    function mlpForward(layers, x) {
  return layers.reduce((a, {W, b, act}) => layerForward(W, b, a, act), x);
}

// add as many layers as you want:
const net = [
  { W: W1, b: b1, act: relu },
  { W: W2, b: b2, act: sigmoid }
];
mlpForward(net, [1.0, -0.5]);  // → same 0.723
