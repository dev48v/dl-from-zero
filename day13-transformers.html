🔨 One Transformer block, in code

    
Embed + position → multi-head attention → residual & norm → feed-forward → residual & norm → stack ×N. ~7 steps. This is exactly the layer repeated inside GPT and BERT — and PyTorch ships it as one line.

    
    
      
  1. 1
    Embed the tokens, then add position
    
        
    // token id → learned vector, then inject order (attention is order-blind)
let X = tokens.map(id => embedding[id]);        // [seqLen, d_model]
X = X.map((vec, pos) => vec.map((v, i) =>
  v + (i % 2 === 0 ? Math.sin(pos / 10000 ** (i / d_model))
                   : Math.cos(pos / 10000 ** ((i - 1) / d_model)))
));   // now "dog bites man" ≠ "man bites dog"
    
          
    2. 

      
  2. 2
    Multi-head self-attention (the Day-12 core)
    
        
    // split d_model into h heads; each head gets its own Q/K/V "view"
function multiHead(X, heads) {
  const outs = [];
  for (let i = 0; i < heads.length; i++) {
    const {Wq, Wk, Wv} = heads[i];
    outs.push(attention(matmul(X, Wq), matmul(X, Wk), matmul(X, Wv)));
  }
  return matmul(concat(outs), Wo);   // combine every head's view
}
const attn = multiHead(X, heads);    // [seqLen, d_model]
    
          
    3. 

      
  3. 3
    Add & LayerNorm (residual #1)
    
        
    // add the block's INPUT back, then normalise each token's features
function layerNorm(row, g = 1, b = 0, eps = 1e-5) {
  const mean = row.reduce((a, v) => a + v, 0) / row.length;
  const varr = row.reduce((a, v) => a + (v - mean) ** 2, 0) / row.length;
  return row.map(v => g * (v - mean) / Math.sqrt(varr + eps) + b);
}
let H = X.map((x, i) => layerNorm(add(x, attn[i])));   // residual + norm
    
          
    4. 

      
  4. 4
    Feed-forward sub-layer (per-token MLP)
    
        
    // a tiny 2-layer MLP applied to EACH token position independently
function feedForward(x) {           // d_model → d_ff (≈4×) → d_model
  const h = relu(add(matmul1d(x, W1), b1));
  return add(matmul1d(h, W2), b2);
}
const ff = H.map(feedForward);      // transforms features token-by-token
    
          
    5. 

      
  5. 5
    Add & LayerNorm (residual #2) → block output
    
        
    // same residual + norm wrapper around the feed-forward output
let out = H.map((h, i) => layerNorm(add(h, ff[i])));
// 'out' has the SAME shape as the input → blocks chain cleanly
return out;   // [seqLen, d_model]
    
          
    6. 

      
  6. 6
    Stack N identical blocks (depth = abstraction)
    
        
    // one block in, same shape out → just feed it forward N times
function transformerBlock(X, p) {
  const a = X.map((x, i) => layerNorm(add(x, multiHead(X, p.heads)[i])));
  const f = a.map(feedForward);
  return a.map((h, i) => layerNorm(add(h, f[i])));
}
let x = embedWithPositions(tokens);
for (const p of blockParams) x = transformerBlock(x, p);  // ×12, ×24, ×96…
    
          
    7. 

      
  7. 7
    In real frameworks: one line (PyTorch)
    
        
    # everything above is already packaged — d_model, n_heads, d_ff, norm, residual
import torch, torch.nn as nn

layer   = nn.TransformerEncoderLayer(d_model=512, nhead=8, dim_feedforward=2048)
encoder = nn.TransformerEncoder(layer, num_layers=12)   # stack of 12 blocks

x   = embed(tokens) + positional_encoding(tokens)       # [seq, batch, 512]
out = encoder(x)     # → contextualised vectors, the heart of GPT / BERT
